const canvas = document.getElementById("gameCanvas"); const ctx = canvas.getContext("2d"); const scoreEl = document.getElementById("score"); const bestEl = document.getElementById("bestScore"); const pauseBtn = document.getElementById("pauseBtn"); const muteBtn = document.getElementById("muteBtn"); const musicFile = document.getElementById("musicFile"); const upBtn = document.getElementById("upBtn"); const downBtn = document.getElementById("downBtn"); const leftBtn = document.getElementById("leftBtn"); const rightBtn = document.getElementById("rightBtn"); const box = 20; const rows = canvas.width / box; let snake = []; let food = {}; let direction = "RIGHT"; let paused = false; let score = 0; let bestScore = Number(localStorage.getItem("bestSnakeScore")) || 0; bestEl.textContent = bestScore; const music = new Audio(); music.loop = true; function randomFood() { food = { x: Math.floor(Math.random() * rows) * box, y: Math.floor(Math.random() * rows) * box }; } function resetGame() { snake = [ { x: 10 * box, y: 10 * box } ]; direction = "RIGHT"; score = 0; scoreEl.textContent = score; randomFood(); } function drawCircle(x, y, color) { ctx.fillStyle = color; ctx.beginPath(); ctx.arc( x + box / 2, y + box / 2, box / 2 - 2, 0, Math.PI * 2 ); ctx.fill(); } function gameOver() { alert("Game Over!\nScore: " + score); resetGame(); } function draw() { if (paused) return; ctx.clearRect( 0, 0, canvas.width, canvas.height ); drawCircle(food.x, food.y, "#ff4d6d"); snake.forEach((part, i) => { drawCircle( part.x, part.y, i === 0 ? "#2ecc71" : "#66d966" ); }); let x = snake[0].x; let y = snake[0].y; if (direction === "LEFT") x -= box; if (direction === "RIGHT") x += box; if (direction === "UP") y -= box; if (direction === "DOWN") y += box; if ( x < 0 || y < 0 || x >= canvas.width || y >= canvas.height ) { gameOver(); return; } for (let i = 0; i < snake.length; i++) { if ( snake[i].x === x && snake[i].y === y ) { gameOver(); return; } } const head = { x, y }; if (x === food.x && y === food.y) { score++; scoreEl.textContent = score; if (score > bestScore) { bestScore = score; bestEl.textContent = bestScore; localStorage.setItem( "bestSnakeScore", bestScore ); } randomFood(); } else { snake.pop(); } snake.unshift(head); } document.addEventListener( "keydown", e => { if ( e.key === "ArrowLeft" && direction !== "RIGHT" ) { direction = "LEFT"; } if ( e.key === "ArrowRight" && direction !== "LEFT" ) { direction = "RIGHT"; } if ( e.key === "ArrowUp" && direction !== "DOWN" ) { direction = "UP"; } if ( e.key === "ArrowDown" && direction !== "UP" ) { direction = "DOWN"; } } ); upBtn.onclick = () => { if (direction !== "DOWN") { direction = "UP"; } }; downBtn.onclick = () => { if (direction !== "UP") { direction = "DOWN"; } }; leftBtn.onclick = () => { if (direction !== "RIGHT") { direction = "LEFT"; } }; rightBtn.onclick = () => { if (direction !== "LEFT") { direction = "RIGHT"; } }; pauseBtn.onclick = () => { paused = !paused; pauseBtn.textContent = paused ? "▶ Resume" : "⏸ Pause"; }; let muted = false; muteBtn.onclick = () => { muted = !muted; music.muted = muted; muteBtn.textContent = muted ? "🔈 Unmute" : "🔊 Mute"; }; musicFile.addEventListener( "change", function () { const file = this.files[0]; if (!file) return; music.src = URL.createObjectURL(file); music.play(); } ); resetGame(); setInterval(draw, 120);
